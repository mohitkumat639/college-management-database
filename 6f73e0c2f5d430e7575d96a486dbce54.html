CREATE TABLE student(
stu_id INT PRIMARY KEY,
stu_name VARCHAR(30),
stu_ph_no INT ,
stu_email VARCHAR(30) ,
stu_dob DATE
);
CREATE TABLE books (
book_id INT PRIMARY KEY ,
book_name VARCHAR(30),
athr_name VARCHAR(30),
book_brn_id INT,
book_price INT
);
CREATE TABLE branch (
brn_id INT PRIMARY KEY,
brn_name VARCHAR(20),
brn_ctg VARCHAR(20)
);
CREATE TABLE user (
usr_id INT PRIMARY KEY,
usr_name VARCHAR(30),
usr_ph_no INT,
usr_email VARCHAR(30)
);
CREATE TABLE issues (
isu_id INT UNIQUE,
isu_date DATE ,
isu_book_id INT ,
isu_stu_id INT
);
CREATE TABLE login (
lgn_id INT UNIQUE NOT NULL,
lgn_usr_id INT,
lgn_usr_name VARCHAR(20),
lgn_pass VARCHAR(18),
lgn_role_id INT
);
CREATE TABLE roles (
rol_id INT PRIMARY KEY,
rol_name VARCHAR(15),
rol_lvl VARCHAR(15)
);
CREATE TABLE permission (
pr_id INT UNIQUE NOT NULL,
pr_name VARCHAR(15),
pr_rol_id INT
);
